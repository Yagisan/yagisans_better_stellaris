# Star Patrol
sstarpatrolbase = {
	potential = {
		has_starbase_size > starbase_outpost
		has_starbase_building = ybs_star_patrol
		exists = owner
		owner = {
			is_gestalt = no
		}
		# Not in colony systems
		not = {
			solar_system = {
				any_system_colony = {
				}
			}
		}
		# 1 per sector
		exists = sector
		sector = {
			any_system_within_border = {
				starbase = {
					NOT = {
						is_starbase_type = sstarpatrolbase
					}
				}
			}
		}
		# Not bordering another empire
		not = {
			solar_system = {
				any_neighbor_system = {
					exists = space_owner
					space_owner = {
						not = {
							is_same_empire = root.owner
						}
					}
				}
			}
		}
	}
	weight_modifier = {
		base = 0
		complex_trigger_modifier = {
			#+75 for each hangar_bay
			trigger = count_starbase_modules
			parameters = {
				type = hangar_bay
			}
			mode = add
			mult = 75
		}
		modifier = {
			add = 1000
			has_starbase_building = ybs_star_patrol
		}
		modifier = {
			add = 150
			# Choke point - more desirable
			solar_system = {
				is_bottleneck_system = yes
			}
		}
		# Not near to an existing Star Patrol
		modifier = {
			factor = 0
			owner = {
				is_ai = yes
			}
			solar_system = {
				closest_system = {
					min_steps = 0
					max_steps = 3
					limit = {
						exists = starbase
						exists = owner
						owner = {
							is_same_empire = root.owner
						}
						starbase = {
							has_starbase_building = ybs_star_patrol
						}
					}
				}
			}
		}
		# want to be near center of empire, not borders
		# has_trade_route = yes
		# num_trade_routes - Counts the number trade routes in the empire.
		modifier = {
			# Useless if cut off or if every owned system has a starbase
			factor = 0
			owner = {
				is_ai = yes
				is_same_empire = root.owner
			}
			solar_system = {
				OR = {
					# No trade routes?
					has_trade_route = no
					# We have a starbase in every system? (really?)
					owner = {
						num_starbases >= trigger:controlled_systems
					}
					# We're isolated. An exclave?
					NOT = {
						any_neighbor_system = {
							exists = owner
							owner = {
								is_same_empire = root.owner
							}
						}
					}
				}
			}
		}
	}
	ai_design = {
		min = {
			base = 0
		}
		ratio = {
			base = 0.2
		}
		buildings = {
			ybs_star_patrol = {
				base = 10
			}
			command_center = {
				base = 5
			}
			cordyceptic_reanimation_facility = {
				base = 3
			}
			offspring_outlook = {
				base = 4
			}
			communications_jammer = {
				base = 1
				modifier = {
					add = 3
					exists = solar_system
					solar_system = {
						is_star_class = sc_black_hole
					}
				}
			# First to shoot often wins.
			target_uplink_computer = {
				base = 3
			}
			curator_think_tank = {
				base = 2
			}
			art_college = {
				base = 2
			}
			trader_proxy_office = {
				base = 2
			}
			salvager_workshop = {
				base = 2
			}
			black_hole_observatory = {
				base = 2
			}
			nebula_refinery = {
				base = 2
			}
			ice_mining_station = {
				base = 2
			}
			dragon_hatchery = {
				base = 2
			}
			disruption_field = {
				base = 1.5
			}
			mercenary_garrison = {
				base = 1.5
			}
			defense_grid = {
				base = 1.5
			}
			resource_silo = {
				base = 1
			}
			hydroponics_bay = {
				base = 1
			}
			# Hyperlane Registrar
			warp_fluctuator = {
				base = 1
			}
			shroud_beacon = {
				base = 1
			}
		}
		modules = {
			hangar_bay = {
				base = 1
				modifier = {
					add = 1000
					NOT = {
						has_starbase_module = hangar_bay
					}
				}
			}
		}
	}
}
