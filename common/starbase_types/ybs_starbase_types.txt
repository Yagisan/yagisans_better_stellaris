# Star Patrol
sstarpatrolbase = {
	potential = {
		has_starbase_size > starbase_outpost
		exists = owner
		owner = {
			is_gestalt = no
		}
		# Not in colony systems
		not = {
			solar_system = {
				any_system_planet = {
					is_colony = yes
				}
			}
		}
		# Not bordering another empire
		not = {
			solar_system = {
				any_neighbor_system = {
					exists = space_owner
					space_owner = {
						not = {
							is_same_value = root.owner
						}
					}
				}
			}
		}
	}
	weight_modifier = {
		base = 0
		complex_trigger_modifier = {
			#+75 for each hangar_bay
			trigger = count_starbase_modules
			parameters = {
				type = hangar_bay
			}
			mode = add
			mult = 75
		}
		modifier = {
			add = 1000
			has_starbase_building = ybs_star_patrol
			exists = owner
		}
	}
	ai_design = {
		min = {
			base = 0
		}
		ratio = {
			base = 0.2
		}
		buildings = {
			ybs_star_patrol = {
				base = 1
				modifier = {
					add = 1000
					NOT = {
						has_starbase_building = ybs_star_patrol
					}
				}
			}
			target_uplink_computer = {
				base = 1
			}
			communications_jammer = {
				base = 1
				modifier = {
					add = 500
					NOT = {
						has_starbase_building = communications_jammer
					}
				}
			}
			disruption_field = {
				base = 1
			}
			nebula_refinery = {
				base = 1
			}
			resource_silo = {
				base = 0.5
			}
			art_college = {
				base = 1
			}
			curator_think_tank = {
				base = 1
			}
			ice_mining_station = {
				base = 1
			}
			dragon_hatchery = {
				base = 1
			}
			hydroponics_bay = {
				base = 1
			}
		}
		modules = {
			hangar_bay = {
				base = 1
				modifier = {
					add = 1000
					NOT = {
						has_starbase_module = hangar_bay
					}
				}
			}
		}
	}
}
