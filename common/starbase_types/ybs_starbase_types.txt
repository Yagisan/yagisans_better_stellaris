# Star Patrol
sstarpatrolbase = {
	potential = {
		has_starbase_size > starbase_outpost
		exists = owner
		owner = {
			is_gestalt = no
		}
		# Not in colony systems
		not = {
			solar_system = {
				any_system_planet = {
					is_colony = yes
				}
			}
		}
		# Not bordering another empire
		not = {
			solar_system = {
				any_neighbor_system = {
					exists = space_owner
					space_owner = {
						not = {
							is_same_value = root.owner
						}
					}
				}
			}
		}
	}
	weight_modifier = {
		base = 0
		complex_trigger_modifier = {
			#+75 for each hangar_bay
			trigger = count_starbase_modules
			parameters = {
				type = hangar_bay
			}
			mode = add
			mult = 75
		}
		modifier = {
			add = 1000
			has_starbase_building = ybs_star_patrol
		}
		modifier = {
			add = 150
			# Choke point - more desirable
			solar_system = {
				is_bottleneck_system = yes
			}
		}
		# Not near to an existing Star Patrol
		modifier = {
			factor = 0
			owner = {
				is_ai = yes
			}
			solar_system = {
				closest_system = {
					min_steps = 0
					max_steps = 3
					limit = {
						exists = starbase
						exists = owner
						owner = {
							is_same_value = root.owner
						}
						starbase = {
							has_starbase_building = ybs_star_patrol
						}
					}
				}
			}
		}
	}
	ai_design = {
		min = {
			base = 0
		}
		ratio = {
			base = 0.2
		}
		buildings = {
			ybs_star_patrol = {
				base = 5
			}
			offspring_outlook = {
				base = 3
			}
			communications_jammer = {
				base = 3
			}
			command_center = {
				base = 3
			}
			deep_space_black_site = {
				base = 2
			}
			transit_hub = {
				base = 2
			}
			curator_think_tank = {
				base = 2
			}
			art_college = {
				base = 2
			}
			trader_proxy_office = {
				base = 2
			}
			salvager_workshop = {
				base = 2
			}
			black_hole_observatory = {
				base = 2
			}
			nebula_refinery = {
				base = 2
			}
			ice_mining_station = {
				base = 2
			}
			dragon_hatchery = {
				base = 2
			}
			disruption_field = {
				base = 1.5
			}
			mercenary_garrison = {

			# First to shoot often wins.

# Listening Post
slisteningpost = {
	potential = {
		has_starbase_size > starbase_outpost
	}
	weight_modifier = {
		base = 0
		complex_trigger_modifier = {
			#+75 for each hangar_bay
			trigger = count_starbase_modules
			parameters = {
				type = hangar_bay
			}
			mode = add
			mult = 75
		}
		modifier = {
			add = 1000
			has_starbase_building = listening_post
		}
		# Next to a neighbour
		modifier = {
			add = 300
			solar_system = {
				any_neighbor_system = {
					exists = owner
					owner = {
						NOT = {
							is_same_value = root.owner
						}
					}
				}
			}
		}
		# Next to a hostile neighbour
		modifier = {
			add = 300
			solar_system = {
				any_neighbor_system = {
					exists = owner
					owner = {
						NOT = {
							is_same_value = root.owner
						}
						relative_power = {
							who = root.owner
							category = all
							value > pathetic
						}
						OR = {
							NOT = {
								has_established_contact = root.owner
							}
							opinion_level = {
								who = root.owner
								level <= poor
							}
							is_rival = root.owner
							is_hostile = root.owner
							is_domineering_to = root.owner
							is_hostile_to = root.owner
							is_disloyal_to = root.owner
							is_unfriendly_to = root.owner
							# Fallen empire
							is_angry_to = root.owner
							# Awakened Empire with low opinion
							# Double up because they are likely coming for us
							is_belligerent_to = root.owner
							root.owner = {
								is_threatened_to = prev
							}
							# They want our land!
							has_claim = root.owner
							# Better safe than sorry
							has_valid_civic = civic_fanatic_purifiers
							has_valid_civic = civic_hive_devouring_swarm
							has_valid_civic = civic_machine_assimilator
							has_valid_civic = civic_machine_terminator
						}
					}
				}
			}
		}
		# Not near to an existing Listening Post
		modifier = {
			factor = 0
			owner = {
				is_ai = yes
			}
			solar_system = {
				closest_system = {
					min_steps = 0
					max_steps = 2
					limit = {
						exists = starbase
						exists = owner
						owner = {
							is_same_value = root.owner
						}
						starbase = {
							has_starbase_building = listening_post
						}
					}
				}
			}
		}
	}
	ai_use_defense_platforms = yes	#AI will not build defense platforms otherwise, except in the capital system
	ai_design = {
		min = {
			base = 0
		}
		ratio = {
			base = 0.2
			modifier = {
				add = 0.1
				exists = owner
				owner = {
					has_ethic = ethic_militarist
				}
			}
			modifier = {
				add = 0.1
				exists = owner
				owner = {
					has_ethic = ethic_fanatic_militarist
				}
			}
		}
		# token followed by priority
		buildings = {
			listening_post = {
				base = 5
			}
			offspring_outlook = {
				base = 3
			}
			communications_jammer = {
				base = 3
			}
			command_center = {
				base = 3
			}
			# First to shoot often wins.
			target_uplink_computer = {
				base = 3
			}
			deep_space_black_site = {
				base = 2
			}
			transit_hub = {
				base = 2
			}
			curator_think_tank = {
				base = 2
			}
			art_college = {
				base = 2
			}
			trader_proxy_office = {
				base = 2
			}
			salvager_workshop = {
				base = 2
			}
			black_hole_observatory = {
				base = 2
			}
			nebula_refinery = {
				base = 2
			}
			ice_mining_station = {
				base = 2
			}
			dragon_hatchery = {
				base = 2
			}
			disruption_field = {
				base = 1.5
			}
			mercenary_garrison = {
				base = 1.5
			}
			defense_grid = {
				base = 1.5
			}
			resource_silo = {
				base = 1
			}
			hydroponics_bay = {
				base = 1
			}
			# Hyperlane Registrar
			warp_fluctuator = {
				base = 1
			}
			shroud_beacon = {
				base = 1
			}
		}
		modules = {
